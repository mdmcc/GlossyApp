CREATE TABLE customer (
  ID int NOT NULL AUTO_INCREMENT,
  customer_email VARCHAR(50) NOT NULL,
  customer_password VARCHAR(50) NOT NULL,
  customer_firstName VARCHAR(25) NOT NULL,
  customer_lastName VARCHAR(30) NOT NULL,
  fb_id VARCHAR(200),
  PRIMARY KEY (ID)
);

CREATE TABLE hairStylist (
  ID int NOT NULL AUTO_INCREMENT,
  hairStylist_email VARCHAR(50) NOT NULL,   
  hairStylist_password VARCHAR(50) NOT NULL,
  hairStylist_firstName VARCHAR(25) NOT NULL,
  hairStylist_lastName VARCHAR(30) NOT NULL,
  fb_id VARCHAR(200),
  PRIMARY KEY (ID)
);

CREATE TABLE appointment (
  ID int NOT NULL AUTO_INCREMENT,
  customer_id int NOT NULL,
  appointment_service_id int NOT NULL,
  customer_dateTime date NOT NULL,
  customer_lng VARCHAR(50) NOT NULL,
  customer_lat VARCHAR(50) NOT NULL,
  hairStylist_id int, 
  appointment_status int,
  -- 0 = active, 1 = inactive, 2 = completed
  -- FOREIGN KEY (customer_id) REFERENCES customer(ID) ON UPDATE RESTRICT ON DELETE RESTRICT,
  FOREIGN KEY (appointment_service_id) REFERENCES appointment_service(ID) ON UPDATE RESTRICT ON DELETE RESTRICT,
  -- FOREIGN KEY (hairStylist_id) REFERENCES hairStyist(ID) ON UPDATE RESTRICT ON DELETE RESTRICT,
  PRIMARY KEY (ID)
);

CREATE TABLE appointment_service (
  ID int NOT NULL AUTO_INCREMENT,
  cut_type int,
  -- 0 = n/a, 1 = simpleCut, 2 = premiumCut
  color_type int,
  -- 0 = n/a, 1 = simpleColor, 2 = premiumColor
  styling_type int,
  -- 0 = n/a, 1 = simpleStyling, 2 = premiumStyling
  PRIMARY KEY (ID)
);
